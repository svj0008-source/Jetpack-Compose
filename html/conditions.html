If


    if (condition) {
  // block of code to be executed if the condition is true
}
Use if to specify a block of code to be executed if a condition is true.

if (20 > 18) {
  println("20 is greater than 18")
}

Else


if (condition) {
  // block of code to be executed if the condition is true
} else {
  // block of code to be executed if the condition is false
}
Use else to specify a block of code to be executed if the condition is false.

val time = 20
if (time < 18) {
  println("Good day.")
} else {
  println("Good evening.")
}
// Outputs "Good evening."

Else if


if (condition1) {
  // block of code to be executed if condition1 is true
} else if (condition2) {
  // block of code to be executed if the condition1 is false and condition2 is true
} else {
  // block of code to be executed if the condition1 is false and condition2 is false
}
Use else if to specify a new condition if the first condition is false.

val time = 22
if (time < 10) {
  println("Good morning.")
} else if (time < 20) {
  println("Good day.")
} else {
  println("Good evening.")
}
// Outputs "Good evening."

When


fun main() {
    val number = 2

    when (number) {
        1 -> println("One")
        2 -> println("Two")
        3 -> println("Three")
        else -> println("Unknown number")
    }
}
Instead of writing many if..else expressions, you can use the when expression, which is much easier to read. 

fun main() {
    val age = 16

    when {
        age < 13 -> println("Child")
        age < 18 -> println("Teenager")
        else -> println("Adult")
    }
}
