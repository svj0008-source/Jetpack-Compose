arrayOf()


   val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")
Arrays are used to store multiple values in a single variable, instead of creating separate variables for each value. To create an array, use the arrayOf() function, and place the values in a comma-separated list inside it

val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")
println(cars[0])
// Outputs Volvo

size


val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")
println(cars.size)
// Outputs 4
To find out how many elements an array have, use the size property

fun main() {
    val fruits = arrayOf("Apple", "Banana", "Orange")

    for (i in 0 until fruits.size) {
        println(fruits[i])
    }
}

for


fun main() {
    for (i in 1..5) {
        println(i)
    }
}
You can loop through the array elements with the for loop, which you will learn even more about in the next chapter.

fun main() {
    val fruits = arrayOf("Apple", "Banana", "Orange")

    for (fruit in fruits) {
        println(fruit)
    }
}
